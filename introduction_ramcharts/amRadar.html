amRadar


settings

  • data : data.frame first column is named “label” (character), other columns are series of values
  • col : character, color(s) of serie(s) hexadecimal like “#00FF00”
  • backTransparency : numeric, background transparency, between 0 and 1
  • type : character, type of radar, must be “polygons” or “circle”
  • pch : character, symbols must be in “round”, “square”, “triangleUp”, “triangleDown”, “triangleLeft”, “triangleRight”, “bubble”,“diamond”, “xError”, “yError”
  • … : see amOptions for more options

Data use in example

data <- data.frame(label = c("A", "Z", "E", "R", "T"),
                   Product1 = c(1, 2, 3, 4, 2), 
                   Product2 = c(2, 8, 1, 1, 0),
                   Product3 = c(1,1,2,2,4))

Basic exemple


amRadar(data)

Change color


amRadar(data, col = c("#0000FF", "#00FF00", "#FF0000"))

backTransparency


amRadar(data, backTransparency = 1)

amRadar(data, backTransparency = c(0, 0.4, 0.6))


type


amRadar(data, type = "circles")

amRadar(data, type = "polygons") #default

pch


amRadar(data,  pch = "triangleRight", backTransparency = 0)


amRadar(data,  pch = "diamond", backTransparency = 0) 


amRadar(data,  pch = "yError", backTransparency = 0)


amRadar(data,  pch = "bubble", backTransparency = 0)

amOptions

legend


amRadar(data,  pch = "bubble", backTransparency = 0, legend = TRUE)
#or
amRadar(data,  pch = "bubble", backTransparency = 0) %>>% amOptions(legend = TRUE)

legendPosition


amRadar(data,  pch = "bubble", backTransparency = 0, legend = TRUE)
#or
amRadar(data,  pch = "bubble", backTransparency = 0) %>>% amOptions(legend = TRUE)

export


amRadar(data,  pch = "bubble", backTransparency = 0, export = TRUE)
#or
amRadar(data,  pch = "bubble", backTransparency = 0) %>>% 
  amOptions(export = TRUE)


exportFormat


amRadar(data,  pch = "bubble", backTransparency = 0, export = TRUE, exportFormat = "CSV")
#or
amRadar(data,  pch = "bubble", backTransparency = 0) %>>% 
  amOptions(export = TRUE, exportFormat = "CSV")

main and creditsPosition


amRadar(data,  pch = "bubble", backTransparency = 0, main = "My Radar",
        mainColor = "#00FF00", mainSize = 25, creditsPosition = "bottom-right")
#or
amRadar(data,  pch = "bubble", backTransparency = 0) %>>% 
  amOptions( main = "My Radar",
        mainColor = "#00FF00", mainSize = 25, creditsPosition = "bottom-right")

theme


amRadar(data,  pch = "bubble", backTransparency = 1, main = "My Radar", theme = "patterns")
#or
amRadar(data,  pch = "bubble", backTransparency = 1) %>>% 
  amOptions( theme = "patterns")
